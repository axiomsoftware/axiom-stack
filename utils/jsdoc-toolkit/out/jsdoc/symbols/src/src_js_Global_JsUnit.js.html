 20 /**
 21  *  @constructor
 22  */
 23 function JsUnitError( msg ){
 24     this.message = msg || "";
 25 }
 26 JsUnitError.prototype.toString = function(){
 27     var msg = this.message;
 28     return this.name + ": " + msg;
 29 }
 30 JsUnitError.prototype = new Error();
 31 JsUnitError.prototype.name = "JsUnitError";
 32 
 33 /**
 34  *  @constructor
 35  */
 36 function AssertionFailedError( msg, stack ){
 37     JsUnitError.call( this, msg );
 38     this.mCallStack = stack;
 39 }
 40 AssertionFailedError.prototype = new JsUnitError();
 41 AssertionFailedError.prototype.name = "AssertionFailedError";

 43 /**
 44  * A set of assert methods.
 45  * @constructor
 46  */
 47 Assert = {};
 48 /**
 49  * Asserts that two values are equal.
 50  * @tparam String msg An optional error message.
 51  * @tparam Object expected The expected value.
 52  * @tparam Object actual The actual value.
 53  * @exception AssertionFailedError Thrown if the expected value is not the
 54  * actual one.
 55  */
 56 Assert.assertEquals = function( msg, expected, actual )
 57 {
 58     if( arguments.length == 2 )
 59     {
 60         actual = expected;
 61         expected = msg;
 62         msg = null;
 63     }
 64     if( expected != actual ){
 65 		Assert.fail( "Expected:[" + expected + "], but was:[" + actual + "]"
 66 				   , /*new CallStack()*/'', msg );
 67 	}
 68 }
 69 /**
 70  * Asserts that a regular expression matches a string.
 71  * @tparam String msg An optional error message.
 72  * @tparam Object expected The regular expression.
 73  * @tparam Object actual The actual value.
 74  * @exception AssertionFailedError Thrown if the actual value does not match
 75  * the regular expression.
 76  * @note This is an enhancement to JUnit 3.8
 77  * @since 1.3
 78  */
 79 Assert.assertMatches = function( msg, expected, actual )
 80 {
 81     if( arguments.length == 2 )
 82     {
 83         actual = expected;
 84         expected = msg;
 85         msg = null;
 86     }
 87     if( expected instanceof RegExp && typeof( actual ) == "string" )
 88     {
 89         if( !actual.match( expected ))
 90             Assert.fail( "RegExp:[" + expected + "] did not match:[" + actual + "]", /*new CallStack()*/undefined, msg );
 91     }
 92     else
 93         Assert.fail( "Expected:[" + expected + "], but was:[" + actual + "]"
 94             , /*new CallStack()*/undefined, msg );
 95 }
 96 /**
 97  * Asserts that a condition is false.
 98  * @tparam String msg An optional error message.
 99  * @tparam String cond The condition to evaluate.
100  * @exception AssertionFailedError Thrown if the evaluation was not false.
101  */
102 Assert.assertFalse = function( msg, cond )
103 {
104     if( arguments.length == 1 )
105     {
106         cond = msg;
107         msg = null;
108     }
109     if( eval( cond ))
110         Assert.fail( "Condition should have failed \"" + cond + "\""
111             , /*new CallStack()*/undefined, msg );
112 }
113 /**
114  * Asserts that two floating point values are equal to within a given tolerance.
115  * @tparam String msg An optional error message.
116  * @tparam Object expected The expected value.
117  * @tparam Object actual The actual value.
118  * @tparam Object tolerance The maximum difference allowed to make equality check pass.
119  * @note This is an enhancement to JUnit 3.8
120  * @exception AssertionFailedError Thrown if the expected value is not within
121  * the tolerance of the actual one.
122  */
123 Assert.assertFloatEquals = function( msg, expected, actual, tolerance)
124 {
125     if( arguments.length == 3 )
126     {
127         tolerance = actual;
128         actual = expected;
129         expected = msg;
130         msg = null;
131     }
132     if(    typeof( actual ) != "number"
133         || typeof( expected ) != "number"
134         || typeof( tolerance ) != "number" )
135     {
136         Assert.fail( "Cannot compare " + expected + " and " + actual
137             + " with tolerance " + tolerance + " (must all be numbers).");
138     }
139 
140     if( Math.abs(expected - actual) > tolerance)
141     {
142         Assert.fail( "Expected:[" + expected + "], but was:[" + actual + "]"
143             , /*new CallStack()*/undefined, msg );
144     }
145 }
146 /**
147  * Asserts that an object is not null.

149  * @tparam Object object The valid object.
150  * @exception AssertionFailedError Thrown if the object is not null.
151  */
152 Assert.assertNotNull = function( msg, object )
153 {
154     if( arguments.length == 1 )
155     {
156         object = msg;
157         msg = null;
158     }
159     if( object === null )
160         Assert.fail( "Object was null.", /*new CallStack()*/undefined, msg );
161 }

216 /**
217  * Asserts that two values are the same.
218  * @tparam String msg An optional error message.
219  * @tparam Object expected The expected value.
220  * @tparam Object actual The actual value.
221  * @exception AssertionFailedError Thrown if the expected value is not the
222  * actual one.
223  */
224 Assert.assertSame = function( msg, expected, actual )
225 {
226     if( arguments.length == 2 )
227     {
228         actual = expected;
229         expected = msg;
230         msg = null;
231     }
232     if( expected === actual )
233         return;
234     else
235         Assert.fail( "Same expected:[" + expected + "], but was:[" + actual + "]"
236             , /*new CallStack()*/undefined, msg );
237 }

246         cond = msg;

251 	}
252 }

269 /**
270  * Fails a test with a give message.
271  * @tparam String msg The error message.
272  * @tparam CallStack stack The call stack of the error.
273  * @tparam String usermsg The message part of the user.
274  * @exception AssertionFailedError Is always thrown.
275  */
276 Assert.fail = function( msg, stack, usermsg ){
277    throw new AssertionFailedError( ( usermsg ? usermsg + " " : "" ) + msg, stack );
